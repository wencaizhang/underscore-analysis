# _.memoize

```js
var memory = function (fn) {
  var cache = {}

  return function () {
    var key = JSON.stringify(Array.prototype.slice.call(arguments));

    if (cache[key]) {
      console.log('缓存：是')
      return cache[key]
    } else {
      console.log('缓存：否')
      return cache[key] = fn.apply(this, arguments)
    }
  }
}

var add = function (a, b) {
  return a + b;
}

var meAdd = memory(add)

console.log(meAdd(1,2,3))
console.log(meAdd(1,2,3))
console.log(meAdd(1,2,3))
console.log(meAdd(1,2,3))
console.log(meAdd(1,2,3))
console.log(meAdd(1,2,3))
```
